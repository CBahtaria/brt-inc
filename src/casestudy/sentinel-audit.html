Key Fixes — Before / After

F-01: Hardcoded credentials removed

// BEFORE — login form pre-filled with production credentials
<input type="text" name="username" value="commander" />
<input type="password" name="password" value="commander123" />

// AFTER — empty inputs, credentials only via environment variables
<input type="text" name="username" placeholder="Username" />
<input type="password" name="password" placeholder="Password" />

F-03: CORS wildcard replaced with allowlist

// BEFORE
header('Access-Control-Allow-Origin: *');

// AFTER — allowlist from environment variable
$allowed = array_filter(explode(',', $_ENV['SENTINEL_CORS_ORIGINS'] ?? ''));
$origin = $_SERVER['HTTP_ORIGIN'] ?? '';
if (in_array($origin, $allowed, true)) {
    header('Access-Control-Allow-Origin: ' . $origin);
    header('Vary: Origin');
}

F-05: Brute-force gate off-by-one corrected

// BEFORE — allowed 6 attempts (>= instead of >)
if ($attempts >= MAX_LOGIN_ATTEMPTS) { lockout(); }

// AFTER
if ($attempts > MAX_LOGIN_ATTEMPTS) { lockout(); }
// MAX_LOGIN_ATTEMPTS = 5, so 6th attempt is correctly blocked
